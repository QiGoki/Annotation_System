# 可自定义模块标注页面设计

## 背景

JSONL数据来源多样，不同项目的JSON结构完全不同：

```json
// 项目A: components 结构
{
  "components": [{"bbox": [...], "text": "...", "type": "...", "children": [...]}],
  "height": 1008,
  "width": 476,
  "image": "..."
}

// 项目B: conversations 结构
{
  "conversations": [{"from": "human", "value": "<image>图中位于 [385, 429, 657, 449] 的元素对应什么内容？"}],
  "images": ["images/id5185_task70_3.jpg"],
  "index": 7,
  "is_completed": true
}
```

传统JSON Path模板化解析无法适配这种多样性。


## 1. 模块注册表（Module Registry）

每个模块声明需要的字段，fieldMapping作为智能推荐函数。

```javascript
const componentRegistry = {
  // === 图片查看器 ===
  ImageViewer: {
    name: '图片查看器',
    icon: '🖼️',
    schema: {
      fields: {
        src: { type: 'string', label: '图片路径', required: true },
        width: { type: 'number', label: '宽度', required: false },
        height: { type: 'number', label: '高度', required: false },
        bboxes: { type: 'array', label: 'BBox列表', required: false }
      }
    },
    // fieldMapping 作为"智能推荐"
    fieldMapping: {
      src: (data) => data.image || data.images?.[0] || data.img,
      width: (data) => data.width,
      height: (data) => data.height,
      bboxes: (data) => {
        // 智能提取：可能是 components、elements、widgets 等
        const arr = data.components || data.elements || data.widgets;
        if (!arr) return [];
        const bboxes = [];
        const extract = (items) => {
          for (const item of items) {
            if (item.bbox) bboxes.push(item.bbox);
            if (item.children) extract(item.children);
          }
        };
        extract(arr);
        return bboxes;
      }
    }
  },

  // === 组件树 ===
  ComponentTree: {
    name: '组件树',
    icon: '🌲',
    schema: {
      fields: {
        items: { type: 'array', label: '组件列表', required: true },
        labelField: { type: 'string', label: '显示文本字段', default: 'text' },
        bboxField: { type: 'string', label: 'BBox字段', default: 'bbox' },
        childrenField: { type: 'string', label: '子元素字段', default: 'children' }
      }
    },
    fieldMapping: {
      items: (data) => data.components || data.elements || data.ui_components
    }
  },

  // === 文字编辑器 ===
  TextEditor: {
    name: '文字编辑',
    icon: '✏️',
    schema: {
      fields: {
        value: { type: 'string', label: '文本值', required: true },
        label: { type: 'string', label: '标签', default: '标注结果' }
      }
    },
    fieldMapping: {
      value: (data) => data.conversations?.[1]?.value || data.annotation || data.result
    }
  },

  // === JSON树 ===
  JsonTree: {
    name: 'JSON树',
    icon: '📋',
    schema: {
      fields: {
        data: { type: 'object', label: '数据对象', required: true }
      }
    },
    fieldMapping: {
      data: (data) => data  // 默认显示整条数据
    }
  }
};
```

### 2.2 自动提取可用字段

```javascript
function extractFields(data, prefix = '') {
  const fields = [];

  for (const [key, val] of Object.entries(data)) {
    const fullPath = prefix ? `${prefix}.${key}` : key;

    if (val === null || val === undefined) continue;

    if (Array.isArray(val)) {
      fields.push({
        path: fullPath,
        type: 'array',
        length: val.length,
        sample: val[0]
      });
    } else if (typeof val === 'object') {
      fields.push({
        path: fullPath,
        type: 'object',
        preview: JSON.stringify(val).slice(0, 50) + '...'
      });
      // 递归提取子字段
      fields.push(...extractFields(val, fullPath));
    } else {
      fields.push({
        path: fullPath,
        type: typeof val,
        sample: val
      });
    }
  }
  return fields;
}

// 结果示例
[
  { path: 'image', type: 'string', sample: '/mnt/xxx/image.jpg' },
  { path: 'components', type: 'array', length: 5 },
  { path: 'components[0].bbox', type: 'array' },
  { path: 'components[0].text', type: 'string' },
  { path: 'width', type: 'number' },
  { path: 'height', type: 'number' }
]
```

## 3. 智能绑定逻辑

### 3.1 核心函数

```javascript
// 智能绑定：fieldMapping 作为默认值，手动绑定可覆盖
function smartBind(moduleType, exampleData) {
  const moduleDef = componentRegistry[moduleType];
  const bindings = {};

  for (const [fieldName, fieldSchema] of Object.entries(moduleDef.schema.fields)) {
    // 1. 先尝试 fieldMapping
    const mappingFn = moduleDef.fieldMapping[fieldName];
    if (mappingFn) {
      const mappedValue = mappingFn(exampleData);
      // 检查映射结果是否有效
      if (mappedValue !== undefined && mappedValue !== null) {
        // 需要确定路径
        const path = findPathByValue(exampleData, mappedValue);
        if (path) {
          bindings[fieldName] = path;
          continue;
        }
      }
    }

    // 2. fieldMapping 失败，尝试智能推断
    const inferred = inferField(fieldName, fieldSchema, exampleData);
    if (inferred) {
      bindings[fieldName] = inferred;
    }
  }

  return bindings;
}

// 通过值反推路径
function findPathByValue(data, targetValue, currentPath = '') {
  for (const [key, val] of Object.entries(data)) {
    const newPath = currentPath ? `${currentPath}.${key}` : key;

    if (val === targetValue) {
      return newPath;
    }

    if (Array.isArray(val)) {
      for (let i = 0; i < val.length; i++) {
        if (val[i] === targetValue) {
          return `${newPath}[${i}]`;
        }
        if (typeof val[i] === 'object') {
          const found = findPathByValue(val[i], targetValue, `${newPath}[${i}]`);
          if (found) return found;
        }
      }
    }

    if (typeof val === 'object' && val !== null) {
      const found = findPathByValue(val, targetValue, newPath);
      if (found) return found;
    }
  }
  return null;
}

// 智能推断字段
function inferField(fieldName, schema, data) {
  const fields = extractFields(data);

  // 精确匹配
  const exact = fields.find(f => f.path === fieldName || f.path.endsWith('.' + fieldName));
  if (exact) return exact.path;

  // 类型匹配
  const matched = fields.find(f => f.type === schema.type);
  return matched?.path || '';
}
```

## 4. UI：手动为主，支持一键推荐

```vue
<!-- 属性面板 - 数据绑定 -->
<template>
  <div class="props-panel">
    <h3>数据绑定</h3>

    <div class="actions">
      <button @click="autoBind" class="btn-auto">
        🤖 智能推荐
      </button>
      <button @click="clearBindings" class="btn-clear">
        清空
      </button>
    </div>

    <div v-for="field in moduleSchema.fields" :key="field.name" class="field-bind">
      <label>
        {{ field.label }}
        <span v-if="field.required" class="required">*</span>
      </label>

      <select v-model="moduleConfig.fieldBindings[field.name]">
        <option value="">-- 选择字段 --</option>
        <optgroup label="推荐字段" v-if="getRecommendedFields(field.name).length">
          <option
            v-for="f in getRecommendedFields(field.name)"
            :key="f.path"
            :value="f.path"
          >
            {{ f.path }} ({{ f.type }})
          </option>
        </optgroup>
        <optgroup label="所有字段">
          <option
            v-for="f in availableFields"
            :key="f.path"
            :value="f.path"
          >
            {{ f.path }} ({{ f.type }})
          </option>
        </optgroup>
      </select>
    </div>

    <!-- 预览绑定结果 -->
    <div class="preview" v-if="hasBindings">
      <h4>数据预览</h4>
      <pre>{{ JSON.stringify(bindPreview, null, 2) }}</pre>
    </div>
  </div>
</template>

<script setup>
function autoBind() {
  // 调用智能绑定，用 fieldMapping 填充
  const bindings = smartBind(props.moduleType, props.exampleData);
  moduleConfig.value.fieldBindings = { ...bindings };
}

function getRecommendedFields(fieldName) {
  const mapping = componentRegistry[props.moduleType].fieldMapping;
  const fn = mapping[fieldName];
  if (!fn) return [];

  const result = fn(props.exampleData);
  if (!result) return [];

  const path = findPathByValue(props.exampleData, result);
  if (path) {
    return availableFields.value.filter(f => f.path === path);
  }
  return [];
}
</script>
```

## 6. 运行时数据解析

```javascript
// 根据 fieldBinding 解析数据
function resolveFieldBinding(data, bindingPath) {
  if (!bindingPath) return undefined;

  const segments = bindingPath.match(/[^.[\]]+|\[\d+\]/g);
  let result = data;

  for (const seg of segments) {
    if (seg.startsWith('[') && seg.endsWith(']')) {
      result = result[parseInt(seg.slice(1, -1))];
    } else {
      result = result[seg];
    }
    if (result === undefined || result === null) return undefined;
  }
  return result;
}

// 批量解析模块数据
function resolveModuleData(moduleConfig, jsonlRow) {
  const result = {};

  for (const [fieldName, bindingPath] of Object.entries(moduleConfig.fieldBindings)) {
    result[fieldName] = resolveFieldBinding(jsonlRow, bindingPath);
  }

  return result;
}
```

## 9. 两种方式的角色

| 方式 | 作用 | 使用场景 |
|------|------|----------|
| **fieldMapping** | 智能默认值 | 新手用户、一键推荐、快速上手 |
| **手动绑定** | 自定义覆盖 | 特殊字段、复杂结构、微调 |

用户可以：
- 点击"🤖 智能推荐"快速填充
- 然后根据需要手动调整个别字段
- 支持清空后完全手动选择

---

## 10. 关键优势

| 优势 | 说明 |
|------|------|
| **多数据源支持** | fieldMapping 支持多种JSON结构 |
| **智能推荐** | fieldMapping 提供一键初始化 |
| **灵活定制** | 手动绑定覆盖默认值 |
| **声明式** | 模块声明需要的字段，注入器自动填充 |
| **解耦** | 模块只关心标准化数据，不关心数据来源 |
