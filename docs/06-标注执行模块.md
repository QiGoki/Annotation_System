# 标注执行模块

## 6.1 功能概述

标注执行模块是标注人员实际进行标注工作的界面，负责：
- 加载任务和配置
- 动态渲染标注页面
- 标注交互操作
- 结果保存
- 任务导航

---

## 6.2 API接口设计

### 6.2.1 标注接口

#### GET /api/tasks/{task_id}

获取任务详情

**响应**：
```json
{
  "code": 200,
  "data": {
    "id": 1,
    "project_id": 1,
    "project_name": "目标检测数据集-2024Q1",
    "project_config": {
      "version": "1.0",
      "dataType": "image",
      "components": [...]
    },
    "data_source": {
      "type": "image",
      "url": "/data/images/001.jpg"
    },
    "status": "doing",
    "annotation": {
      "id": 1,
      "result_json": {...},
      "created_at": "2024-01-15T10:00:00Z"
    }
  }
}
```

#### POST /api/tasks/{task_id}/annotate

保存标注结果

**请求头**：`Authorization: Bearer <token>`

**请求体**：
```json
{
  "result_json": {
    "rect_1": [
      {"x": 100, "y": 200, "width": 150, "height": 80, "label": "person"}
    ],
    "classify_1": "person",
    "input_1": "目标清晰"
  }
}
```

**响应**：
```json
{
  "code": 200,
  "message": "保存成功"
}
```

#### POST /api/tasks/{task_id}/submit

提交标注结果

**响应**：
```json
{
  "code": 200,
  "message": "提交成功",
  "data": {
    "next_task_id": 2
  }
}
```

#### GET /api/tasks/{task_id}/next

获取下一条任务（用于快速跳转）

**查询参数**：
| 参数 | 类型 | 必填 | 说明 |
|------|------|------|------|
| project_id | int | 否 | 项目ID |
| status | string | 否 | 任务状态 |

**响应**：
```json
{
  "code": 200,
  "data": {
    "task_id": 2,
    "has_next": true
  }
}
```

### 6.3.2 标注执行页面

```vue
<!-- views/annotation/AnnotationRunner.vue -->
<template>
  <div class="annotation-runner">
    <!-- 顶部导航 -->
    <div class="header">
      <div class="project-info">
        <span class="project-name">{{ projectName }}</span>
        <el-tag :type="statusType">{{ taskStatus }}</el-tag>
      </div>
      <div class="user-info">
        <span>{{ username }}</span>
        <el-button link @click="handleLogout">退出</el-button>
      </div>
    </div>

    <!-- 主内容区 -->
    <div class="main-content">
      <!-- 左侧：数据展示 -->
      <div class="data-panel">
        <div class="data-container" ref="dataContainerRef">
          <!-- 图像展示 -->
          <template v-if="dataType === 'image'">
            <img
              v-if="dataSource.url"
              :src="dataSource.url"
              class="data-image"
              @load="onImageLoad"
            />
            <!-- 标注层 -->
            <canvas
              ref="canvasRef"
              class="annotation-canvas"
              @mousedown="handleCanvasMouseDown"
              @mousemove="handleCanvasMouseMove"
              @mouseup="handleCanvasMouseUp"
            />
          </template>

          <!-- 文本展示 -->
          <template v-else-if="dataType === 'text'">
            <div class="data-text" @mouseup="handleTextSelect">
              {{ dataSource.content }}
            </div>
          </template>
        </div>
      </div>

      <!-- 右侧：标注表单 -->
      <div class="form-panel">
        <el-scrollbar>
          <el-form
            :model="formData"
            :rules="rules"
            ref="formRef"
            label-position="top"
          >
            <template v-for="component in components" :key="component.id">
              <component
                :is="getComponent(component.type)"
                v-model="formData[component.id]"
                :label="component.label"
                :required="component.required"
                :attributes="component.attributes"
                :validation="component.validation"
                :data-source="dataSource"
                :canvas-ref="canvasRef"
                @update:modelValue="handleComponentChange"
              />
            </template>
          </el-form>
        </el-scrollbar>
      </div>
    </div>

    <!-- 底部工具栏 -->
    <div class="footer">
      <div class="nav-buttons">
        <el-button :disabled="!prevTaskId" @click="goToTask(prevTaskId)">
          上一条
        </el-button>
        <el-button :disabled="!nextTaskId" @click="goToTask(nextTaskId)">
          下一条
        </el-button>
      </div>
      <div class="action-buttons">
        <el-button @click="handleAutoSave" :loading="saving">
          自动保存中...
        </el-button>
        <el-button @click="handleSave" type="primary" :loading="saving">
          保存
        </el-button>
        <el-button @click="handleSubmit" type="success" :loading="saving">
          提交
        </el-button>
      </div>
    </div>
  </div>
</template>

<script setup lang="ts">
import { ref, computed, onMounted, onUnmounted, watch } from 'vue'
import { useRoute, useRouter } from 'vue-router'
import { useUserStore } from '@/stores/user'
import { getTask, saveAnnotation, submitAnnotation, getNextTask } from '@/api/task'
import { getComponent } from './components/AnnotationComponentMap'
import { ElMessage } from 'element-plus'

const route = useRoute()
const router = useRouter()
const userStore = useUserStore()

// DOM引用
const dataContainerRef = ref<HTMLElement>()
const canvasRef = ref<HTMLCanvasElement>()
const formRef = ref()

// 状态
const loading = ref(false)
const saving = ref(false)
const taskId = computed(() => Number(route.params.id))
const username = computed(() => userStore.user?.username)

// 数据
const projectName = ref('')
const dataType = ref<'image' | 'text'>('image')
const dataSource = ref<any>({})
const components = ref<any[]>([])
const formData = ref<Record<string, any>>({})
const taskStatus = ref('pending')
const prevTaskId = ref<number | null>(null)
const nextTaskId = ref<number | null>(null)

// 计算属性
const statusType = computed(() => {
  const map: Record<string, any> = {
    pending: 'info',
    doing: 'warning',
    completed: 'success'
  }
  return map[taskStatus.value] || 'info'
})

const rules = computed(() => {
  const result: Record<string, any[]> = []
  components.value.forEach(comp => {
    if (comp.required) {
      result[comp.id] = [
        { required: true, message: comp.validation?.message || `${comp.label}为必填项`, trigger: 'change' }
      ]
    }
  })
  return result
})

// 生命周期
onMounted(() => {
  loadTask()
  setupAutoSave()
  setupKeyboard()
})

onUnmounted(() => {
  clearAutoSave()
  removeKeyboard()
})

// 方法
const loadTask = async () => {
  loading.value = true
  try {
    const res = await getTask(taskId.value)
    const data = res.data

    projectName.value = data.project_name
    dataType.value = data.project_config.dataType
    dataSource.value = data.data_source
    components.value = data.project_config.components
    taskStatus.value = data.status

    // 加载已有标注或初始化
    if (data.annotation?.result_json) {
      formData.value = data.annotation.result_json
    } else {
      initFormData()
    }

    // 获取相邻任务
    const nextRes = await getNextTask(taskId.value)
    nextTaskId.value = nextRes.data.task_id || null

  } catch (e: any) {
    ElMessage.error('加载任务失败')
  } finally {
    loading.value = false
  }
}

const initFormData = () => {
  formData.value = {}
  components.value.forEach(comp => {
    const defaultValue = comp.attributes?.defaultValue
    if (defaultValue !== undefined) {
      formData.value[comp.id] = defaultValue
    } else if (comp.type === 'image_classify' && comp.attributes?.multi) {
      formData.value[comp.id] = []
    } else {
      formData.value[comp.id] = null
    }
  })
}

const handleSave = async () => {
  await formRef.value?.validate()
  saving.value = true
  try {
    await saveAnnotation(taskId.value, { result_json: formData.value })
    taskStatus.value = 'doing'
    ElMessage.success('保存成功')
  } catch (e: any) {
    ElMessage.error('保存失败')
  } finally {
    saving.value = false
  }
}

const handleSubmit = async () => {
  await formRef.value?.validate()
  saving.value = true
  try {
    await submitAnnotation(taskId.value, { result_json: formData.value })
    taskStatus.value = 'completed'
    ElMessage.success('提交成功')

    // 跳转到下一条
    if (nextTaskId.value) {
      router.push(`/annotate/${nextTaskId.value}`)
    } else {
      ElMessage.info('已是最后一条')
    }
  } catch (e: any) {
    ElMessage.error('提交失败')
  } finally {
    saving.value = false
  }
}

const goToTask = (id: number) => {
  if (id) {
    router.push(`/annotate/${id}`)
  }
}

// 自动保存
let autoSaveTimer: number | null = null

const setupAutoSave = () => {
  autoSaveTimer = window.setInterval(async () => {
    if (taskStatus.value !== 'completed' && Object.keys(formData.value).length > 0) {
      try {
        await saveAnnotation(taskId.value, { result_json: formData.value })
        console.log('自动保存成功')
      } catch (e) {
        console.error('自动保存失败', e)
      }
    }
  }, 30000) // 30秒
}

const clearAutoSave = () => {
  if (autoSaveTimer) {
    clearInterval(autoSaveTimer)
  }
}

// 快捷键
const setupKeyboard = () => {
  window.addEventListener('keydown', handleKeyDown)
}

const removeKeyboard = () => {
  window.removeEventListener('keydown', handleKeyDown)
}

const handleKeyDown = (e: KeyboardEvent) => {
  // Ctrl+S 保存
  if (e.ctrlKey && e.key === 's') {
    e.preventDefault()
    handleSave()
  }
  // Ctrl+Enter 提交
  if (e.ctrlKey && e.key === 'Enter') {
    e.preventDefault()
    handleSubmit()
  }
  // 左箭头 上一条
  if (e.key === 'ArrowLeft' && prevTaskId.value) {
    goToTask(prevTaskId.value)
  }
  // 右箭头 下一条
  if (e.key === 'ArrowRight' && nextTaskId.value) {
    goToTask(nextTaskId.value)
  }
}
</script>
```

```vue
  <el-form-item :label="label" :required="required" :prop="validation ? componentId : ''">

      <div class="rect-list" v-if="rects.length > 0">
        <div
          v-for="(rect, index) in rects"
          :key="index"
          class="rect-item"
          :class="{ active: activeIndex === index }"
          @click="activeIndex = index"
        >
          <span>目标 {{ index + 1 }}</span>
          <el-button type="danger" link @click.stop="removeRect(index)">删除</el-button>
        </div>
      </div>

      </div>
    </div>

  canvasRef: HTMLCanvasElement
  validation?: any
}>()

// 内部ID用于表单校验

  get: () => props.modelValue || [],

  }
```

## 6.5 快捷操作

### 6.5.1 支持的快捷键

| 快捷键 | 功能 |
|--------|------|
| Ctrl + S | 保存 |
| Ctrl + Enter | 提交 |
| ← | 上一条 |
| → | 下一条 |
| Esc | 取消当前操作 |

### 6.5.2 自动保存

- 默认30秒自动保存一次
- 页面离开时自动保存
- 网络断开时暂存本地，恢复后上传

## 6.6 状态管理

### 6.6.1 任务状态流转

```
pending → doing → completed
  ↑          ↓
  └──────────┘ (可回退)
```

### 6.6.2 前端状态

| 状态 | 说明 |
|------|------|
| 未加载 | 任务数据未加载 |
| 加载中 | 正在请求任务数据 |
| 已加载 | 任务数据已显示 |
| 保存中 | 正在保存标注结果 |
| 已保存 | 标注已保存 |
| 提交成功 | 任务已完成 |
